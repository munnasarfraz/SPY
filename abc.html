import csv # Define the input string with indices you want to match input_indices = "0,2,3,4,5,6,7,7" # Convert it into a list of integers input_indices = list(map(int, input_indices.split(','))) # Initialize an empty list to store the corresponding values result_values = [] # Open and read the CSV file with open('test.csv', mode='r') as file: reader = csv.reader(file) # Skip the header if there is one next(reader) # Iterate through each row in the CSV file for row in reader: # Assuming the first column is index and the second column is value index = int(row[0]) value = row[1] # Check if the index is in the input_indices list if index in input_indices: result_values.append(value) # Print the result as a comma-separated string print(', '.join(result_values))
